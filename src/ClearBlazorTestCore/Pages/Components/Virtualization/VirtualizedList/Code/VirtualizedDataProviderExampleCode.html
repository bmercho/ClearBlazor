    VirtualizedList<Feed> Virtualize = null!;

    record Feed(int ImageId, string Title, string Message);

    List<Feed> _feeds = new List<Feed>();

    protected override void OnInitialized()
    {
        base.OnInitialized();
        for (int i = 0; i < 10000; i++)
        {
            _feeds.Add(new Feed((i + 1) % 1000, $"Record #{i}", "This is a message"));
        }
    }

    async Task<(int totalNumItem, IEnumerable<Feed> items)> GetData(ClearBlazor.DataProviderRequest request)
    {
        await Task.CompletedTask;
        return (_feeds.Count(), _feeds.Skip(request.StartIndex).Take(request.Count));
    }
