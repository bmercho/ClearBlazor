@code {
    NonVirtualizedList<Feed> Virtualize = null!;
    record Feed(int ImageId, string Title, string Message);

    List<Feed> _feeds = new List<Feed>();

    protected override void OnInitialized()
    {
        base.OnInitialized();
        for (int i = 0; i < 100; i++)
        {
            _feeds.Add(new Feed((i + 1) % 1000, $"Record #{i}", GetMessage(i)));
        }
    }

    async Task OnGoto10Start()
    {
        await Virtualize.GotoIndex(10, Alignment.Start);
    }

    async Task OnGoto50Center()
    {
        await Virtualize.GotoIndex(50, Alignment.Center);
    }

    async Task OnGoto70End()
    {
        await Virtualize.GotoIndex(70, Alignment.End);
    }

    async Task OnAddNewItem()
    {
        var index = _feeds.Count;
        _feeds.Add(new Feed((index + 1) % 1000, $"Record #{index}", GetMessage(index)));
        await Virtualize.Refresh();
    }

    async Task OnGotoEnd()
    {
        await Virtualize.GotoEnd();
    }

    async Task OnGotoStart()
    {
        await Virtualize.GotoStart();
    }

    async Task OnAddNewItemGotoEndIfAtEnd()
    {
        var atEnd = await Virtualize.AtEnd();

        var index = _feeds.Count;
        _feeds.Add(new Feed((index + 1) % 1000, $"Record #{index}", GetMessage(index)));
        await Virtualize.Refresh();
        if (atEnd)
            await Virtualize.GotoEnd();
    }
