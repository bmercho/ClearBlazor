@using ClearBlazor
<WrapPanel Height=500 Width=500 BorderThickness=1>
    <Spinner Size=Size.VerySmall/>
    <Spinner Size=Size.Small/>
    <Spinner Size=Size.Normal/>
    <Spinner Size=Size.Large/>
    @if (_loading)
    {
        <Spinner Size=Size.VeryLarge/>
    }
</WrapPanel>

<Button OnClicked=@OnClick>Load</Button>

@code {
    bool _loading = false;

    private async Task OnClick()
    {
        _loading = true;

        await Task.Delay(5000);

        _loading = false;
    }
