<StackPanel Orientation="Orientation.Landscape">
    <IconButton Icon=@Icon IconColour=@Colour Colour=@Colour OnClicked=@Clicked/>
    <TextBlock VerticalAlignment="Alignment.Center" Margin="10,0,0,0">Alarm is @(toggled ? "On" : "Off")</TextBlock>
</StackPanel>

@code {
    public string Icon { get; set; } = Icons.Material.Filled.AlarmOff;
    public Color Colour { get; set; } = Color.Primary;
    private bool toggled = false;

    private void Clicked()
    {
        toggled = !toggled;
        Icon = toggled == true ? Icons.Material.Filled.AlarmOn : Icons.Material.Filled.AlarmOff;
        Colour = toggled == true ? Color.Error : Color.Primary;
    }
}
